// State
let history = [];
let currentNode = null;
let arrangementType = null; // 'new' or 'reassess'
let materialityChanged = null; // true/false — only for reassess path
let checkboxCounter = 0;

const LANDING = '__landing__';
const REASSESS_CHANGE = '__reassess_change__';

function getNode(id) {
  if (id === 'start') return tree;
  return nodes[id] || null;
}

function getResult(id) {
  return results[id] || null;
}

function isResult(id) {
  return id && id.startsWith('result_');
}

function render() {
  const container = document.getElementById('treeContainer');

  if (!currentNode || currentNode === LANDING) {
    document.getElementById('progressFill').style.width = '0%';
    document.getElementById('breadcrumb').innerHTML = '';
    document.getElementById('breadcrumb').style.display = 'none';
    document.getElementById('backBtn').style.display = 'none';
    document.querySelector('.progress-bar').style.display = 'none';
    renderLanding(container);
    return;
  }

  if (currentNode === REASSESS_CHANGE) {
    document.getElementById('breadcrumb').style.display = '';
    document.getElementById('backBtn').style.display = '';
    document.querySelector('.progress-bar').style.display = '';
    document.getElementById('progressFill').style.width = '92%';
    document.getElementById('backBtn').disabled = false;
    renderReassessChange(container);
    return;
  }

  document.getElementById('breadcrumb').style.display = '';
  document.getElementById('backBtn').style.display = '';
  document.querySelector('.progress-bar').style.display = '';

  const totalSteps = 9;
  const progress = Math.min((history.length / totalSteps) * 100, 90);
  document.getElementById('progressFill').style.width = progress + '%';

  const bc = document.getElementById('breadcrumb');
  const items = [arrangementType === 'reassess' ? 'Reassessment' : 'New arrangement'];
  history.forEach(h => { if (h.answerLabel) items.push(h.answerLabel); });
  bc.innerHTML = items.map((item, i) => {
    const isLast = i === items.length - 1;
    return `<span class="breadcrumb-item ${isLast ? 'active' : ''}">${item}</span>${!isLast ? '<span class="breadcrumb-sep">›</span>' : ''}`;
  }).join('');

  document.getElementById('backBtn').disabled = history.length === 0;

  if (isResult(currentNode)) {
    renderResult(container, getResult(currentNode));
  } else {
    const node = getNode(currentNode);
    if (node) renderQuestion(container, node);
  }
}

function setArrangementType(type) {
  arrangementType = type;
  materialityChanged = null;
  history = [];
  currentNode = 'start';
  render();
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

function setMaterialityChanged(changed) {
  materialityChanged = changed;
  const lastEntry = history[history.length - 1];
  const resultNode = lastEntry?.nextResult;
  history.push({ node: REASSESS_CHANGE, answerLabel: changed ? 'Materiality changed' : 'Same materiality' });
  currentNode = resultNode;
  render();
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

function answer(nextId, label) {
  if (arrangementType === 'reassess' && isResult(nextId) && materialityChanged === null) {
    history.push({ node: currentNode, answerLabel: label, nextResult: nextId });
    currentNode = REASSESS_CHANGE;
    render();
    window.scrollTo({ top: 0, behavior: 'smooth' });
    return;
  }
  history.push({ node: currentNode, answerLabel: label });
  currentNode = nextId;
  render();
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

function goBack() {
  if (history.length === 0) {
    currentNode = LANDING;
    arrangementType = null;
    materialityChanged = null;
    render();
    window.scrollTo({ top: 0, behavior: 'smooth' });
    return;
  }
  const prev = history.pop();
  if (currentNode === REASSESS_CHANGE || prev.nextResult) {
    materialityChanged = null;
  }
  currentNode = prev.node;
  render();
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

function restart() {
  history = [];
  currentNode = LANDING;
  arrangementType = null;
